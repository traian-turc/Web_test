namespace Struct_01
{
    public partial class Form1 : Form
    {
        public Form1()
        {
            InitializeComponent();
        }
        struct semnal_ac
        {
            public int u_n;
            public int i_n;
        };
        static int nr_max;
        static int val_max = 1024;
        int val_u;
        int val_i;
        static semnal_ac[] valori = new semnal_ac[0];
        System.Drawing.Graphics Desen;
        System.Drawing.Pen Creion_albastru;
        System.Drawing.Pen Creion_rosu;
        private void Form1_Paint(object sender, PaintEventArgs e)
        {
            nr_max = 1 + this.Width / 20;
            Desen = this.CreateGraphics();
            Creion_albastru = new System.Drawing.Pen(System.Drawing.Color.Blue);
            Creion_rosu = new System.Drawing.Pen(System.Drawing.Color.Red);
        }

        private void button1_Click(object sender, EventArgs e)
        {
            int i, x;
            Array.Resize(ref valori, nr_max + 1);
            Desen.Clear(this.BackColor);
            System.Random n = new System.Random();
            i = 0;
            for (x = 20; x <= this.Width; x += 20, i++)
            {
                val_u = n.Next(val_max);
                valori[i].u_n = val_u;
                val_i = n.Next(val_max);
                valori[i].i_n = val_i;
            }
            this.button2.Enabled = true;
            this.label1.Text = "Au fost generate " + nr_max.ToString()+" valori! ";
        }

        private void button2_Click(object sender, EventArgs e)
        {
            int i, x, x_v, y_u, y_i, y_u_v, y_i_v;
            x_v = 0;
            val_u = valori[0].u_n;
            y_u_v = this.Height - 50 - System.Convert.ToInt16(((double)val_u / (double)val_max) * (this.Height - 50));
            val_i = valori[0].i_n;
            y_i_v = this.Height - 50 - System.Convert.ToInt16(((double)val_i / (double)val_max) * (this.Height - 50));
            i = 1;
            for (x = 20; x <= this.Width; x += 20, i++)
            {
                val_u = valori[i].u_n;
                y_u = this.Height - 50 - System.Convert.ToInt16(((double)val_u / (double)val_max) * (this.Height - 50));
                Desen.DrawLine(Creion_albastru, x_v, y_u_v, x, y_u);
		Desen.DrawEllipse(Creion_albastru, x-2, y_u-2, 4, 4);
                y_u_v = y_u;
                val_i = valori[i].i_n;
                y_i = this.Height - 50 - System.Convert.ToInt16(((double)val_i / (double)val_max) * (this.Height - 50));
                Desen.DrawLine(Creion_rosu, x_v, y_i_v, x, y_i);
		Desen.DrawEllipse(Creion_rosu, x-2, y_i-2, 4, 4);
                x_v = x;
                y_i_v = y_i;
            }
            this.button2.Enabled = false;
            this.label1.Text = "Generati noi valori!";
        }

    namespace Struct_04
{
    public partial class Form1 : Form
    {
        public Form1()
        {
            InitializeComponent();
        }
        System.Drawing.Pen creion1, creion2;
        System.Drawing.Graphics desen;
        System.Random aleat;
        int nrc,x,y,r,xi,yi,ri,xci,yci,alfai;
        double alfair;
        private void Form1_Load(object sender, EventArgs e)
        {
            creion1 = new System.Drawing.Pen(System.Drawing.Color.Blue);
            creion2 = new System.Drawing.Pen(System.Drawing.Color.Red);         
            aleat = new System.Random();
            nrc = 30;
            x = 100;
            y = 30;

        }

        private void button1_Click(object sender, EventArgs e)
        {
            
            desen = this.CreateGraphics();
            desen.Clear(this.BackColor);
            r = (this.Width - 30)/4;
            alfai = aleat.Next(360);
            alfair = 2 * Math.PI * alfai / 360;
            ri=10+aleat.Next(r/5);
            xci = x + r + System.Convert.ToInt32(r * Math.Cos(alfair));
            yci = y + r - System.Convert.ToInt32(r * Math.Sin(alfair));
