        public bool compare(Point a, Point b)
        {
            return a.X <= b.X && a.Y <= b.Y;
        }

        // functia alege culoarea urmatoarei secvente din linie
        public int set_color (int ID_color)
        {
            if (ID_color == 0)
            {
                return 1;
            }
            else
                return 0;
        }

        public void Liniute(ref Graphics _draw, Point start, Point end, int lineWidth, int units, double firstLineStart, Color[] colors, int ID_color)
        {
            bool directie = compare(start, end);
            Point unit = new Point((end.X - start.X) / units, (end.Y - start.Y) / units);
            Point aux = new Point(start.X + (int)(unit.X * firstLineStart), start.Y + (int)(unit.Y * firstLineStart));
            ID_color = set_color(ID_color);
            // vom desena o linie de la punctul start pana la punctul aux; prima secventa
            _draw.DrawLine(new Pen(colors[ID_color], (float)LineWidth), start, aux);
            start = aux;
            // se itereaza pana cand nu mai putem intoduce secvente
            while (compare(start, end) == directie)
            {
                Point aux2 = new Point(start.X + unit.X, start.Y + unit.Y);
                // se verifica daca se mai poate introduce o secventa
                if (compare(aux2, end) == directie)
                {
                    aux = new Point(start.X + unit.X, start.Y + unit.Y);
                    ID_color = set_color(ID_color);
                    _draw.DrawLine(new Pen(colors[ID_color], (float)lineWidth), start, aux);
                    start = aux;
                }
                else
                {
                    // se deseneaza ultima secventa
                    ID_color = set_color(ID_color);
                    _draw.DrawLine(new Pen(colors[ID_color], (float)lineWidth), start, end);
                    return;
                }
            }
        }
