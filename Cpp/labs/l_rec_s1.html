        Afisarea primelor 90 numere naturale intr-un tabel cu 9 coloane. Alinierea numerelor se face la stanga 

    #include "stdafx.h"
#include < iostream >
using namespace std;

int main(void)
{
	int i=0;
	int c=1;//coloana
	const int nr_max=90;
	cout <<" \n\tNumerele sunt:\n";
	while (i <= nr_max)
	{
		while((c <= 7) && (i <= nr_max)) {
			cout << i << "\t";
			i+=1;
			c++;	
		}
	cout << "\n";
	c=1;	
	}
	cin.ignore();
	cin.get();
	return 0;
}

        Afisati primele numere naturale sub forma unui tabel cu 7 coloane si 20 de randuri. Alinierea numerelor se va face la dreapta.

{

	while (r<=20)
	{
		while(c <= 7){
			if(i > 99)
			cout << " ";
			if((i > 9)&&(i < 100))
			cout <<"  ";
			if(i < 10)
			cout <<"   ";
			cout << i << " ";
			i+=1;
			c++;	
		}
	cout<<"\n";
	c=1;	
	r++;	
	}

	cin.get();

}


        Se cere un numar dupa care se testeza daca este un numar prim. 

    #include "stdafx.h"
#include < iostream >
using namespace std;

int main(void)
{
	int n=1;
	cout << " \n\Ce numar vreti sa verificam?: ";
	cin >> n;
	int prim=1;
	for (int i=2;i < n; i++)
	{
		if(n%i==0)
		prim=0;
	}
	if (prim==1)
		cout << "Numarul " << n << " este prim";
	else
		cout << "Numarul " << n << " nu este prim";
	cin.ignore();
	cin.get();
	return 0;
}

        Afisati primele 100 de numerele prime intr-un tabel cu 10 coloane 

    #include "stdafx.h"
#include < iostream >
using namespace std;

int main(void)
{
int k=0;
int n=1;
int kmax=100; // numarul max de numere prime

    while(k < kmax) {
		int prim=1;
		for (int i=2; i < n; i++)
		{
			if(n%i==0)
			prim=0;
		}
		if (prim==1){
			cout << n << "\t";
			k++;
			if(k%10==0)
				cout << "\n";
		}
		n++;
    }
	cin.ignore();
	cin.get();
	return 0;
}

        Metoda II sub forma de X

int main(void)
{
	int i,j;
	for(j=1; j <= 25;j++){
		for (i=1;i < j;i++)
			cout<<" ";
		cout << j;
		for (i=25;i > 0;i--){
			if (i > j)
				cout << " ";
			else
				cout << "\b";
		}
		cout << 26-j << "\n";
	}
	cin.ignore();
	cin.get();
	return 0;
}
