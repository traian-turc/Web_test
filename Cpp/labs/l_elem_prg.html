     Afisare binara fara a folosi afis_bin() 

#include "stdafx.h"
#include < iostream >
#include < windows.h >
using namespace std;

int main(void)
{
int nr;
cout << "Introduceti un numar intreg de maxim 255:";
	cin >> nr;
	cout << " Numarul: " << nr << " in binar este:";
	cout << nr/128;
	cout << (nr-(nr/128)*128)/64;
	cout << (nr-(nr/64)*64)/32;
	cout << (nr-(nr/32)*32)/16;
	cout << (nr-(nr/16)*16)/8;
	cout << (nr-(nr/8)*8)/4;
	cout << (nr-(nr/4)*4)/2;
	cout << (nr-(nr/2)*2)/1;
	cin.ignore();
	cin.get();
	return 0;
}
