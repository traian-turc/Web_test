void nr_pare()
{
	int n;
	cout << "Dati n:";
	cin >> n;
	for(int i=1;i <=n;++i)
		if(i%2==0) cout << i <<" ";
}

void min()
{
	int n,x,min=10000;
	cout << "Cate nr cititi de la tastatura?:";
	cin >> n;
	for(int i=1;i <=n ;++i)
	{
		cout << "Dati nr:";
		cin >> x;
		if(x < min) min=x;
	}
	cout << "Minimul este:" << min;

}

void divcumsin()
{
	int m,n;
	cout << "Dati m:";
	cin >> m;
	cout << "Dati n:";
	cin >> n;
	for(int i=10000; i <=99999; ++i)
		if(i%m==0&&i%n==0)
			cout << i << " ";
}
