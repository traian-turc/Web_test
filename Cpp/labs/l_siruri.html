        {
            if (!this.serialPort1.IsOpen)
                this.serialPort1.Open();

            this.serialPort1.Write("A0");

            String txt = this.serialPort1.ReadExisting();
            int l = txt.Length;
            if (l > 3)

                this.label1.Text = val_s;

                }
