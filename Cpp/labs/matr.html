    // Programul utilizeaza o matrice de dimensiunea n x m
// Elementelor matricii li se atribuite valori intregi aleatoare intre 0 si 100
#include "stdafx.h"
#include < iostream >
#include < string >

using namespace std ;

int main(void)
{
	system("TITLE Matrice de dimensiunea n x m ");// Titlul ferestrei consola
	system("COLOR F9"); // Fundal alb caractere albastre
	int i,j,n,m;
	cout <<" \n\  Introduceti valoarea lui n (maxim 10):";
	cin >> n;
	cout << " \n\  Introduceti valoarea lui m (maxim 10):";
	cin >> m;
	const int nmax=10;
	const int mmax=10;
	int matr[nmax][mmax],nr;
	for (j=0; j < m; j++) {
		for (i=0;i < n;i++){
		matr[i][j]= 100*rand()/RAND_MAX ; // pentru a genera numere intre 0-100
		}
	}
	cout << "\n\n\  Matricea este:  \n\n";
	for (j=0; j < m; j++) {
		cout << "\n" ;
		for (i=0;i < n;i++){
			cout << "  matr[" << i << "][" << j << "]=" << matr[i][j] << "\t";
		}
	}
	cout << "\n\n\  Totaluri:  \n\n";
	for (i=0; i < n; i++) {
		int sum=0;
		for (j=0;j < m;j++){
			sum=sum+matr[i][j];
		}
		cout << "  Suma[" << i << "]=" << sum <<"\t";
	}
	cin.ignore();
	cin.get();
	return 0;
}
