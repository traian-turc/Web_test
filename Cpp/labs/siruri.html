        Sterge subsirul din sir 

    // Programul cere un sir si un subsir de caractere .
// Programul sterge subsirul din sir .
// Se folosesc functiile

// Functia strstr(s1, s2) – returneaza pozitia unde apare subsirul s2 in sirul s1
// Functia strcpy(s1, s2) – copieaza sirul s2 in sirul s1
#include "stdafx.h"
#include < iostream >
#include < string >
using namespace std;

int main(void)
{
system("TITLE Stergere subsir din sir ");
system("COLOR F9");

cout << "\n\n\tIntroduceti sirul = ";
cin >> s1;
cout << "\n\tIntroduceti subsirul = ";
cin >> s2;

p=strstr(s1,s2);

        Ordonare alfabetica cuvinte 

    // Programul cere n cuvinte, le ordoneaza alfabetic apoi le afiseaza.
// Se folosesc functiile
// Functia strcmp(s1,s2) – returneaza valoarea 0 daca sirul s1 este egal cu sirul
// s2, valoarea 1 daca sirul s1 este mai mare decat sirul s2 sau -1 daca sirul s1
// este mai mic decat sirul s2
// Functia strcpy(s1, s2) – copieaza sirul s2 in sirul s1
#include "stdafx.h"
#include < iostream >
#include < string >
using namespace std;

int main(void)
{
system("TITLE Ordonare alfabetica cuvinte ");
system("COLOR F9");
char cuvinte[10][25],temp[25];
int i,n,gasit=1;
cout << "\n\n\tIntroduceti numarul de cuvinte :";
cin >> n;
for(i=0;i < n; i++){
	cout << "\n\tcuvantul " << i << " : ";
	cin >> cuvinte[i];
}
while(gasit==1){
gasit=0;
for(i=0;i < n-1; i++)
 if(strcmp(cuvinte[i],cuvinte[i+1]) > 0){
	strcpy(temp, cuvinte[i]);
	strcpy(cuvinte[i],cuvinte[i+1]);
	strcpy(cuvinte[i+1],temp);
	gasit=1;
 }
}
cout << "\n\n\tCuvintele ordonate alfabetic sunt :\n\t";

for(i=0; i < n; i++)
	cout << " \n\n\t: "<< cuvinte[i];
